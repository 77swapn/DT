// Main processing function async function processDownload(url) { // Hide button, show loader and update UI downloadBtn.style.display = 'none'; loader.style.display = 'block'; statusText.style.color = '#add8e6'; statusText.innerText = 'Extracting video, please wait...'; // 🛑 REPLACE THESE WITH YOUR NEW CREDENTIALS const apiKey = 'YOUR_NEW_RAPIDAPI_KEY_HERE'; const apiHost = 'instagram-downloader-download-instagram-stories-videos4.p.rapidapi.com'; // Note: If you switch to a different API provider, update the apiHost and apiUrl accordingly const options = { method: 'GET', headers: { 'x-rapidapi-key': apiKey, 'x-rapidapi-host': apiHost, 'Content-Type': 'application/json' } }; try { const apiUrl = `https://${apiHost}/convert?url=${encodeURIComponent(url)}`; const response = await fetch(apiUrl, options); // 🔍 NEW: Catch bad HTTP responses before trying to parse JSON if (!response.ok) { const errorText = await response.text(); throw new Error(`API HTTP Error: ${response.status} - ${errorText}`); } const data = await response.json(); console.log("Full API Response:", data); // Check your browser console to see the exact structure // Pruned search function to find all unique media URLs in the JSON (carousel-safe) function findAllLinksInResponse(obj, urls = []) { if (!obj) return urls; if (Array.isArray(obj)) { for (let item of obj) { findAllLinksInResponse(item, urls); } } else if (typeof obj === 'object') { let keysToExtract = ['download_link', 'video_url', 'url']; for (let key of keysToExtract) { if (typeof obj[key] === 'string' && obj[key].startsWith('http')) { const val = obj[key]; if (!urls.includes(val) && !val.includes('instagram-downloader') && !val.includes('rapidapi') && !val.includes('/150x150/') && !val.includes('/s150x150/')) { urls.push(val); } } } for (let key in obj) { if (key.toLowerCase().includes('thumb') || key.toLowerCase().includes('preview') || key.toLowerCase().includes('profile')) { continue; } findAllLinksInResponse(obj[key], urls); } } return urls; } const finalDownloadLinks = findAllLinksInResponse(data); if (finalDownloadLinks && finalDownloadLinks.length > 0) { statusText.innerText = `Found ${finalDownloadLinks.length} items. Starting download...`; for (let i = 0; i < finalDownloadLinks.length; i++) { const isVideo = finalDownloadLinks[i].includes('.mp4') || finalDownloadLinks[i].includes('video'); const fileExt = isVideo ? 'mp4' : 'jpg'; await forceDownload(finalDownloadLinks[i], `instagram_media_${i + 1}.${fileExt}`, i + 1, finalDownloadLinks.length); } statusText.innerText = `All ${finalDownloadLinks.length} downloads finished!`; setTimeout(resetUI, 5000); } else { alert('Could not extract any media links. The API structure may have changed. Check the browser console.'); resetUI(); } } catch (err) { alert('Network error or API limit reached. Check the browser console for exact details.'); console.error("Download Error:", err); // This will now print the exact HTTP error resetUI(); } }
